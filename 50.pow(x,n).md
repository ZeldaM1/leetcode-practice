class Solution {
public:
    double myPow(double x, int n) {
        if(n>0){
        double m=1;
        for(int i=0;i<n;i++){
            m=x*m;
        }
        return m;}
      else  if(n<0){
            n=-n;
            double m=1;
        for(int i=0;i<n;i++){
            m=x*m;
        }
            m=1/m;
        return m;
        }
       else if(n=0&&x!=0)
        {
            double m=1;
         return m;
        }
        
    }
};
