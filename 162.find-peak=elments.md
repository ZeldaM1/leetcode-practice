class Solution {
public:
    int findPeakElement(vector<int>& nums) {
     int low = 0,high = nums.size()-1;    
        while(low < high)  //当两者相等时，  
        {    
            int mid = (low+high)/2;       
            if(nums[mid] < nums[mid+1])    
                low = mid+1;  //注意为什么要+1，因为mid位置一定不是峰，而low=mid+1有可能是峰  
            else   
                high = mid;      //注意这里为什么不加+，因为high=mid有可能是峰  
        }    
            
        return low;   
    }  
};
